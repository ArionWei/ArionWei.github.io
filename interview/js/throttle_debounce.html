    # 函数节流与防抖

    # 函数防抖(debounce)

    在事件被触发一定时间后再执行回调，如果在这个一定时间内又被触发，则重新计时。

    <input type="text" class="debounce" />

<script>
  // 模拟发送请求
  function sendAjax(msg) {
    console.log('ajax', msg);
  }

  // 获取dom
  var inputDom = document.querySelector('.debounce')

  // 定义防抖函数
  function debounce(fun, delay) {
    let timer
    return function () {
      var _this = this
      var args = arguments
      if (timer) clearTimeout(timer);
      timer = setTimeout(function () {
        fun.apply(_this, args)
      }, delay)
    }
  }

  let ajax = debounce(sendAjax, 500)

  inputDom.addEventListener('input', function (e) {
    ajax(e.target.value)
  })
</script>

    # 函数节流(throttle)

    规定在一个单位时间内，只能触发一次函数。如果这个单位时间内触发多次函数，只有一次生效。

    function throttle(fun, delay) {
  let flag = true
  return function () {
    let _this = this
    let args = arguments
    if (!flag) return false
    flag = false
    setTimeout(function () {
      fun.apply(_this, args)
      flag = true
    }, delay)
  }
}
